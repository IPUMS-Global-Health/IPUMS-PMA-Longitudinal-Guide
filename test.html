coef shows the estimated population proportion

_low and _upp show the lower and upper bounds of a 95% confidence interval 
library(srvyr)

dat %>% 
  as_survey_design(weight = PANELWEIGHT) %>% 
  filter(CP_1 < 90 & CP_2 < 90) %>% 
  summarise(
    survey_mean(
      CP_1 * CP_2,
      vartype = "ci",
      level = 0.95,
      proportion = TRUE,
      prop_method = "logit"
    )
  )
